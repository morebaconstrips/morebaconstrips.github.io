Key Genie

I found the SAT values by using the Z3 library in the following python script

from z3 import *

# Create Z3 integer variables for param_1, param_2, param_3, and param_4
param_1 = BitVec('param_1', 11)
param_2 = BitVec('param_2', 11)
param_3 = BitVec('param_3', 11)
param_4 = BitVec('param_4', 11)

# Create a Z3 solver
solver = Solver()

# Add constraints based on your conditions
solver.add(param_1 < 1000)
solver.add(param_2 < 1000)
solver.add(param_3 < 1000)
solver.add(param_4 < 1000)

solver.add(param_1 >= 0)
solver.add(param_2 >= 0)
solver.add(param_3 >= 0)
solver.add(param_4 >= 0)

uVar1 = (param_1 % 100) / 10
uVar2 = (param_3 % 100) / 10

solver.add(param_1 / 100 < uVar1)
solver.add(uVar1 < param_1 % 10)
solver.add(param_2 % param_1 == 0)

cond1 = param_3 / 100 == param_2 / 100 - param_1 / 100
cond2 = uVar2 == (param_2 % 100) / 10 - uVar1
cond3 = param_3 % 10 == param_2 % 10 - param_1 % 10

solver.add(cond1)
solver.add(cond2)
solver.add(cond3)

cond4 = (param_3 / 100 % 2) == 0
cond5 = (uVar2 % 2) == 0
cond6 = (param_3 % 10 % 2) == 0

solver.add(cond4)
solver.add(cond5)
solver.add(cond6)

solver.add(param_4 % 10 == (param_4 / 100) ^ (param_4 % 100) / 10)

# Check if there exists a solution
if solver.check() == sat:
